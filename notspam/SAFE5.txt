domtab={

	activeClass:'active', // class of current link

	printID:'domtabprintview', // id of the print all link
	showAllLinkText:'show all content', // text for the print all link
	prevNextIndicator:'doprevnext', // class to trigger prev and next links
	prevNextClass:'prevnext', // class of the prev and next list
	prevLabel:'previous', // HTML content of the prev link
	nextLabel:'next', // HTML content of the next link
	prevClass:'prev', // class for the prev link
	nextClass:'next', // class for the next link
	init:function(){
		var temp;
		if(!document.getElementById || !document.createTextNode){return;}
		var tempelm=document.getElementsByTagName('div');		
		for(var i=0;i<tempelm.length;i++){
			if(!domtab.cssjs('check',tempelm[i],domtab.tabClass)){continue;}
			domtab.initTabMenu(tempelm[i]);
			domtab.removeBackLinks(tempelm[i]);
			if(domtab.cssjs('check',tempelm[i],domtab.prevNextIndicator)){
				domtab.addPrevNext(tempelm[i]);
			}
			domtab.checkURL();
		}
		if(document.getElementById(domtab.printID) 
		   && !document.getElementById(domtab.printID).getElementsByTagName('a')[0]){
			var newlink=document.createElement('a');
			newlink.setAttribute('href','#');
			domtab.addEvent(newlink,'click',domtab.showAll,false);
			newlink.onclick=function(){return false;} // safari hack
			newlink.appendChild(document.createTextNode(domtab.showAllLinkText));
			document.getElementById(domtab.printID).appendChild(newlink);
		}
	},
	checkURL:function(){
		var id;
		var loc=window.location.toString();
		loc=/#/.test(loc)?loc.match(/#(\w.+)/)[1]:'';
		if(loc==''){return;}
		var elm=document.getElementById(loc);
		if(!elm){return;}
		var parentMenu=elm.parentNode.parentNode.parentNode;
		parentMenu.currentSection=loc;
		parentMenu.getElementsByTagName(domtab.contentElements)[0].style.display='none';
		domtab.cssjs('remove',parentMenu.getElementsByTagName('a')[0].parentNode,domtab.activeClass);
		var links=parentMenu.getElementsByTagName('a');
		for(i=0;i<links.length;i++){
			if(!links[i].getAttribute('href')){continue;}
			if(!/#/.test(links[i].getAttribute('href').toString())){continue;}
			id=links[i].href.match(/#(\w.+)/)[1];
			if(id==loc){
				var cur=links[i].parentNode.parentNode;
				domtab.cssjs('add',links[i].parentNode,domtab.activeClass);
				break;
			}
		}
		domtab.changeTab(elm,1);
		elm.focus();
		cur.currentLink=links[i];
		cur.currentSection=loc;
	},
	showAll:function(e){
		document.getElementById(domtab.printID).parentNode.removeChild(document.getElementById(domtab.printID));
		var tempelm=document.getElementsByTagName('div');		
		for(var i=0;i<tempelm.length;i++){
			if(!domtab.cssjs('check',tempelm[i],domtab.tabClass)){continue;}
			var sec=tempelm[i].getElementsByTagName(domtab.contentElements);
			for(var j=0;j<sec.length;j++){
				sec[j].style.display='block';
			}
		}
		var tempelm=document.getElementsByTagName('ul');		
		for(i=0;i<tempelm.length;i++){
			if(!domtab.cssjs('check',tempelm[i],domtab.prevNextClass)){continue;}
			tempelm[i].parentNode.removeChild(tempelm[i]);
			i--;
		}
		domtab.cancelClick(e);
	},
	addPrevNext:function(menu){
		var temp;
		var sections=menu.getElementsByTagName(domtab.contentElements);
		for(var i=0;i<sections.length;i++){
			temp=domtab.createPrevNext();
			if(i==0){
				temp.removeChild(temp.getElementsByTagName('li')[0]);
			}
			if(i==sections.length-1){
				temp.removeChild(temp.getElementsByTagName('li')[1]);
			}
			temp.i=i; // h4xx0r!
			temp.menu=menu;
			sections[i].appendChild(temp);
		}
	},
	removeBackLinks:function(menu){
		var links=menu.getElementsByTagName('a');
		for(var i=0;i<links.length;i++){
			if(!domtab.backToLinks.test(links[i].href)){continue;}
			links[i].parentNode.removeChild(links[i]);
			i--;
		}
	},
	initTabMenu:function(menu){
		var id;
		var lists=menu.getElementsByTagName('ul');
		for(var i=0;i<lists.length;i++){
			if(domtab.cssjs('check',lists[i],domtab.listClass)){
				var thismenu=lists[i];
				break;
			}
		}
		if(!thismenu){return;}
		thismenu.currentSection='';
		thismenu.currentLink='';
		var links=thismenu.getElementsByTagName('a');
		for(i=0;i<links.length;i++){
			if(!/#/.test(links[i].getAttribute('href').toString())){continue;}
			id=links[i].href.match(/#(\w.+)/)[1];
			if(document.getElementById(id)){
				domtab.addEvent(links[i],'click',domtab.showTab,false);
				links[i].onclick=function(){return false;} // safari hack
				domtab.changeTab(document.getElementById(id),0);

		}
		id=links[0].href.match(/#(\w.+)/)[1];
		if(document.getElementById(id)){
			domtab.changeTab(document.getElementById(id),1);
			thismenu.currentSection=id;
			thismenu.currentLink=links[0];


			domtab.cssjs('add',links[0].parentNode,domtab.activeClass);
		}
	},
	createPrevNext:function(){
		// this would be so much easier with innerHTML, darn you standards fetish!
		var temp=document.createElement('ul');
		temp.className=domtab.prevNextClass;
		temp.appendChild(document.createElement('li'));
		temp.getElementsByTagName('li')[0].appendChild(document.createElement('a'));
		temp.getElementsByTagName('a')[0].setAttribute('href','#');
		temp.getElementsByTagName('a')[0].innerHTML=domtab.prevLabel;
		temp.getElementsByTagName('li')[0].className=domtab.prevClass;
		temp.appendChild(document.createElement('li'));
		temp.getElementsByTagName('li')[1].appendChild(document.createElement('a'));
		temp.getElementsByTagName('a')[1].setAttribute('href','#');
		temp.getElementsByTagName('a')[1].innerHTML=domtab.nextLabel;
		temp.getElementsByTagName('li')[1].className=domtab.nextClass;
		domtab.addEvent(temp.getElementsByTagName('a')[0],'click',domtab.navTabs,false);
		domtab.addEvent(temp.getElementsByTagName('a')[1],'click',domtab.navTabs,false);
		// safari fix
		temp.getElementsByTagName('a')[0].onclick=function(){return false;}
		temp.getElementsByTagName('a')[1].onclick=function(){return false;}
		return temp;
	},
	navTabs:function(e){
		var li=domtab.getTarget(e);
		var menu=li.parentNode.parentNode.menu;
		var count=li.parentNode.parentNode.i;
		var section=menu.getElementsByTagName(domtab.contentElements);
		var links=menu.getElementsByTagName('a');
		var othercount=(li.parentNode.className==domtab.prevClass)?count-1:count+1;
		section[count].style.display='none';
		domtab.cssjs('remove',links[count].parentNode,domtab.activeClass);
		section[othercount].style.display='block';
		domtab.cssjs('add',links[othercount].parentNode,domtab.activeClass);
		var parent=links[count].parentNode.parentNode;
		parent.currentLink=links[othercount];
		parent.currentSection=links[othercount].href.match(/#(\w.+)/)[1];
		domtab.cancelClick(e);
	},
	changeTab:function(elm,state){
		do{
			elm=elm.parentNode;
		} while(elm.nodeName.toLowerCase()!=domtab.contentElements)
		elm.style.display=state==0?'none':'block';
	},
	showTab:function(e){
		var o=domtab.getTarget(e);
		if(o.parentNode.parentNode.currentSection!=''){
			domtab.changeTab(document.getElementById(o.parentNode.parentNode.currentSection),0);
			domtab.cssjs('remove',o.parentNode.parentNode.currentLink.parentNode,domtab.activeClass);
		}
		var id=o.href.match(/#(\w.+)/)[1];
		o.parentNode.parentNode.currentSection=id;
		o.parentNode.parentNode.currentLink=o;
		domtab.cssjs('add',o.parentNode,domtab.activeClass);
		domtab.changeTab(document.getElementById(id),1);
		document.getElementById(id).focus();
		domtab.cancelClick(e);
	},
/* helper methods */
	getTarget:function(e){
		var target = window.event ? window.event.srcElement : e ? e.target : null;
		if (!target){return false;}
		if (target.nodeName.toLowerCase() != 'a'){target = target.parentNode;}
		return target;
	},
	cancelClick:function(e){
		if (window.event){
			window.event.cancelBubble = true;
			window.event.returnValue = false;
			return;
		}
		if (e){
			e.stopPropagation();
			e.preventDefault();
		}
	},
	addEvent: function(elm, evType, fn, useCapture){
		if (elm.addEventListener) 
		{
			elm.addEventListener(evType, fn, useCapture);
			return true;
		} else if (elm.attachEvent) {
			var r = elm.attachEvent('on' + evType, fn);
			return r;
		} else {
			elm['on' + evType] = fn;
		}
	},
	cssjs:function(a,o,c1,c2){
		switch (a){
			case 'swap':
				o.className=!domtab.cssjs('check',o,c1)?o.className.replace(c2,c1):o.className.replace(c1,c2);
			break;
			case 'add':
				if(!domtab.cssjs('check',o,c1)){o.className+=o.className?' '+c1:c1;}
			break;
			case 'remove':
				var rep=o.className.match(' '+c1)?' '+c1:c1;
				o.className=o.className.replace(rep,'');
			break;
			case 'check':
				var found=false;
				var temparray=o.className.split(' ');
				for(var i=0;i<temparray.length;i++){
					if(temparray[i]==c1){found=true;}
				}
				return found;
			break;
		}
	}
}

Some of those games were really great, but most of them were not really that enjoyable. So in an effort to create the longest and most useful list of the best free games, I decided to create this user suggested list.<br /><br /><br /><p align="left">So, I propose this to you, I am going to list around 20 of what I think are some of the best free games below with their respective screenshots. If you play a free game that you think other people might like, post a comment with the name or URL of the game. I'll add the best of those games directly into this post, so people don't have to browse all the comments to find what they like.</p><br /><br /><p align="left">So without further ado, here is the "beta" version of a never-ending list of all the best free games on the internet;</p><br /><br /><p align="left"><br /></p><p align="center"><strong><a href="http://www.blogger.com/main=04Within_a_Deep_Forest&amp;sub=01About">Within A Deep Forest </a></strong></p><p align="center"><a href="http://nifflas.ni2.se/index.php?main=04Within_a_Deep_Forest&amp;sub=01About"><img src="http://www.buttermouth.com/games/withinadeepforest.gif" border="0" height="300" width="400" /></a> </p><br /><br /><p align="center">From the site: "Two attempts have been made to build a bomb, capable of freezing the entire world. The first was a failure, and resulted in a living blue ball. The second attempt was a success. Now the clock is ticking, and the world is in grave danger. That is, unless...</p><br /><br /><p align="center">Within a Deep Forest is a freeware 2D platform game for Windows featuring challenging gameplay, beautiful music, an evil doctor, and a deep forest."</p><br /><br /><p align="center"><strong><a href="http://www.planetwolfenstein.com/files/files.shtml">Wolfenstein Enemy Territory</a></strong></p><p align="center"><a href="http://www.planetwolfenstein.com/files/files.shtml"><img src="http://www.buttermouth.com/games/wolfenstein.gif" border="0" height="300" width="400" /></a></p><br /><br /><p align="center">From the site: "Wolfenstein: Enemy Territory was originally planned as an add-on for Return to Castle Wolfenstein, but soon became it's own game, a separate entity with full single player, multiplayer, AI opponents and teammates! Well that didn't happen and what we got, while less than planned, is amazing...and <strong>FREE</strong>! Wolfenstein: Enemy Territory is a free, stand-alone multiplayer only product, which means it's for online play only, but all you need is what you download. "</p><br /><br /><p align="center"><strong><a href="http://www.addictinggames.com/ngame.html">The N-Game</a> </strong></p><p align="center"><a href="http://www.addictinggames.com/ngame.html"><img src="http://www.buttermouth.com/games/ngame.gif" border="0" height="296" width="400" /></a></p><br /><br /><p align="center">From the site: "Golden cubes await capture as you guide your ninja-like icon through treacherous territory. Watch out for mines, lasers, and heat-seeking missiles." </p><br /><br /><p align="center"><a href="http://www.warsow.net/?page=download"><strong>Warsow</strong></a></p><p align="center"><a href="http://www.warsow.net/?page=download"><strong><img src="http://www.buttermouth.com/games/warsow.gif" height="300" width="400" /></strong></a></p><br /><br /><p align="center">From the site: "Warsow is a free standalone first person shooter game for Windows and Linux. It is based on the Qfusion 3D engine (a modification of the Quake 2 GPL engine), and aimed on the competitive scene, or the e-sports community." </p><br /><br /><p align="center"><a href="http://icculus.org/neverball/"><strong>Neverball</strong></a></p><p align="center"><a href="http://icculus.org/neverball/"><strong><img src="http://www.buttermouth.com/games/neverball.gif" height="300" width="400" /></strong></a></p><br /><br /><p align="center">From the site: " Tilt the floor to roll a ball through an obstacle course before time runs out. Neverball is part puzzle game, part action game, and entirely a test of skill."</p><br /><br /><p align="center"><a href="http://smw.72dpiarmy.com/"><strong>Super Mario War </strong></a></p><p align="center"><a href="http://smw.72dpiarmy.com/"><strong><img src="http://www.buttermouth.com/games/supermariowar.gif" height="300" width="400" /></strong> </a></p><br /><br /><p align="center">From the site: "<strong>Super Mario War</strong> is a Super Mario multiplayer game. The goal is to stomp as many other Marios as possible to win the game. It's a tribute to Nintendo and the game Mario War by Samuele Poletto." </p><br /><br /><p align="center"><strong><a href="http://fretsonfire.sourceforge.net/">Frets on Fire </a></strong></p><p align="center"><a href="http://fretsonfire.sourceforge.net/"><strong><img src="http://www.buttermouth.com/games/fof.gif" height="300" width="400" /></strong></a></p><br /><br /><p align="center">Frets on Fire is a "Guitar Hero" clone for the PC. You use your keyboard as the guitar by htting F1-F5 while strumming with the Enter key. </p><br /><br /><p align="center">You can download songs to use alongside Frets on Fire here; <a href="http://fretsonfirelinks.wikidot.com/guitar-hero-songs">Click Here to See a List of Songs </a></p><br /><br /><p align="center"><a href="http://www.flightgear.org/"><strong>Flight Gear Simulation </strong></a></p><p align="center"><a href="http://www.flightgear.org/"><strong><img src="http://www.buttermouth.com/games/flightgear.gif" border="0" height="300" width="400" /></strong></a></p><br /><br /><p align="center">From the site: "FlightGear is an open-source, multi-platform flight simulator." This game is similar to Microsoft's Flight Simulator X.</p><br /><br /><div style="text-align: center;"><br /><a href="http://www.kqzyfj.com/click-3173236-10275193" target="_top"><br /><img src="http://www.awltovhc.com/image-3173236-10275193" alt="Unlimited Video Game Rentals - Start Now!" border="0" height="250" width="250" /></a><br /></div><br /><br /><br /><p align="center"><a href="http://www.urbanterror.net/news.php"><strong>Urban Terror </strong></a></p><p align="center"><a href="http://www.urbanterror.net/news.php"><strong><img src="http://www.buttermouth.com/games/urbanterror.gif" border="0" height="300" width="400" /></strong></a></p><br /><br /><p align="center">From the site: "Urban Terror could best be described as a Hollywood tactical shooter; it is realism based to a certain extent (environments/weapons/player models), but also goes by the motto "fun over realism" (fast gameplay and lots of action). This combination of reality and action results in a very unique, enjoyable and addictive game."</p><br /><br /><p align="center"><a href="http://www.fileplanet.com/166516/160000/fileinfo/F.E.A.R.-Combat-%5BFREE-Multiplayer%5D"><strong>F.E.A.R. Combat </strong></a></p><p align="center"><a href="http://www.fileplanet.com/166516/160000/fileinfo/F.E.A.R.-Combat-%5BFREE-Multiplayer%5D"><strong><img src="http://www.buttermouth.com/games/fear.gif" border="0" height="300" width="400" /></strong> </a></p><br /><br /><p align="center">From the site: "<em>F.E.A.R.&#8482; (First Encounter Assault Recon)</em> is an intense combat experience with rich atmosphere and a deeply intense paranormal storyline presented entirely in first-person. Be the hero in your own spine-tingling epic of action, tension, and terror." </p><br /><br /><p align="center"><a href="http://www.secretmaryo.org/"><strong>Secret Maryo</strong></a></p><p align="center"><a href="http://www.secretmaryo.org/"><img src="http://www.buttermouth.com/games/maryo.gif" height="300" width="400" /></a></p><br /><br /><p align="center">From the site: "Secret Maryo Chronicles is an <strong>Open Source two-dimensional platform game</strong> with a style designed similar to classic sidescroller games." </p><br /><br /><p align="center"><a href="http://www.xenoage.com/bp2k6/en/downloads.php"><strong>Slam Soccer 2006 </strong></a></p><p align="center"><a href="http://www.xenoage.com/bp2k6/en/downloads.php"><strong><img src="http://www.buttermouth.com/games/slamsoccer.gif" border="0" height="300" width="400" /></strong></a></p><br /><br /><p align="center">From the site: "Slam Soccer 2006 is a funny football game in 3D-comic-style - and it's for free!"</p><br /><br /><p align="center"><a href="http://www.trackmanianations.com/indexUk.php"><strong>Trackmania Nations </strong></a></p><p align="center"><a href="http://www.trackmanianations.com/indexUk.php"><strong><img src="http://www.buttermouth.com/games/trackmanianations.gif" border="0" height="300" width="400" /></strong></a></p><br /><br /><p align="center">From the site: "For the first time in the history of eSport, a video game has been specially developed for the Electronic Sports World Cup and is being offered free of charge to the players of the entire planet so they may commence training for the qualifiers due to take place in 53 countries on Friday 27th of January. The Grand Final of ESWC will be held in Paris from June 28th to July the 2nd , in front of 40,000 people, where the best players from each country will battle it out for a prize of 400,000 dollars."</p><p align="center"></p><br /><br /><p align="center"><a href="http://www.americasarmy.com/"><strong>America's Army </strong></a></p><p align="center"><a href="http://www.americasarmy.com/"><strong><img src="http://www.buttermouth.com/games/americaarmy.gif" border="0" height="300" width="400" /></strong> </a></p><br /><br /><p align="center">From the site: " In the <em>America's Army</em> game, players are bound by Rules of Engagement (ROE) and grow in experience as they navigate challenges in teamwork-based, multiplayer, force versus force operations. In the game, as in the Army, accomplishing missions requires a team effort and adherence to the seven Army Core Values."</p><p align="center"></p><br /><br /><p align="center"><a href="http://rakion.softnyx.net/download/client.asp"><strong>Rakion</strong></a></p><p align="center"><a href="http://rakion.softnyx.net/download/client.asp"><strong><img src="http://www.buttermouth.com/games/rakion.gif" border="0" height="300" width="400" /></strong> </a></p><br /><br /><p align="center">From the site: "Rakion is an 'on-line strategy game'. The game focuses on intense sword on sword battles with RPG charcters and settings, along with the strategic battle plans to provide the ultimate battle experience." </p><br /><br /><p align="center">In other words, it's a free MMORPG.</p><p align="center"><br /></p><p align="center"></p><p align="center"><strong><a href="http://www.openarena.ws/">Open Arena </a></strong></p><p align="center"><a href="http://www.openarena.ws/"><strong><img src="http://www.buttermouth.com/games/openarena.gif" border="0" height="300" width="400" /></strong></a><strong></strong> </p><br /><br /><p align="center">From the site: "OpenArena is an open-source content package for Quake III Arena licensed under the GPL, effectively creating a free stand-alone game. <strong>You do not need Quake III Arena to play this game.</strong>" This game is effectively a new and free clone of the quake games.</p><p align="center"></p><p class="style7" align="center"><br /></p><p class="style7" align="center"><span style="font-size:130%;"><span style="font-weight: bold;">Old Skool Games</span></span> </p><p align="center"></p><br /><br /><p align="center"><span class="style2"><a href="http://takegame.com/arcade/htm/chipschallenge.htm"><span style="font-weight: bold;">Chips Challenge</span> </a></span></p><p align="center"><span class="style1"><a href="http://takegame.com/arcade/htm/chipschallenge.htm"><img src="http://www.buttermouth.com/games/chipschallenge.gif" border="0" height="300" width="400" /></a></span></p><br /><br /><p align="center">A fun puzzle game that involves finding keys so tha main character, "Chip", can collect all the computer chips.</p><p align="center"><br /></p><p align="center"></p><p align="center"><a href="http://www.jezzball.org/"><strong>JezzBall</strong></a></p><p align="center"><a href="http://www.jezzball.org/"><img src="http://www.buttermouth.com/games/jezzball.gif" border="0" height="300" width="400" /></a></p><br /><br /><p align="center">Use your right mouse button to swith from vertical dividers to horizontal dividers. Then use your left mouse button to "shoot" these dividers across the screen, in hopes of clearing at least 75% of the screen area.</p><p align="center"></p><br /><br /><p align="center"><strong><a href="http://ski.ihoc.net/">SkiFree</a></strong></p><p align="center"><strong><a href="http://ski.ihoc.net/"><img src="http://www.buttermouth.com/games/skifree.gif" border="0" height="300" width="400" /></a></strong> </p><br /><br /><p align="center">Control a skiier with you mouse. Move it around to move the skiier, and use the mouse buttons to jump higher. Watch out for snowboarders and sasquatch.</p><p align="center"></p><p align="center"></p><br /><br /><p align="center"><strong>Do you know of a game that should be included on this list?

domtab={

	printID:'domtabprintview', // id of the print all link
	showAllLinkText:'show all content', // text for the print all link
	prevNextIndicator:'doprevnext', // class to trigger prev and next links
	prevNextClass:'prevnext', // class of the prev and next list
	prevLabel:'previous', // HTML content of the prev link
	nextLabel:'next', // HTML content of the next link
	prevClass:'prev', // class for the prev link
	nextClass:'next', // class for the next link
	init:function(){
		var temp;
		if(!document.getElementById || !document.createTextNode){return;}
		var tempelm=document.getElementsByTagName('div');		
		for(var i=0;i<tempelm.length;i++){
			if(!domtab.cssjs('check',tempelm[i],domtab.tabClass)){continue;}
			domtab.initTabMenu(tempelm[i]);
			domtab.removeBackLinks(tempelm[i]);
			if(domtab.cssjs('check',tempelm[i],domtab.prevNextIndicator)){
				domtab.addPrevNext(tempelm[i]);
			}
			domtab.checkURL();
		}
		if(document.getElementById(domtab.printID) 
		   && !document.getElementById(domtab.printID).getElementsByTagName('a')[0]){
			var newlink=document.createElement('a');
			newlink.setAttribute('href','#');
			domtab.addEvent(newlink,'click',domtab.showAll,false);
			newlink.onclick=function(){return false;} // safari hack
			newlink.appendChild(document.createTextNode(domtab.showAllLinkText));
			document.getElementById(domtab.printID).appendChild(newlink);
		}
	},
	checkURL:function(){
		var id;
		var loc=window.location.toString();
		loc=/#/.test(loc)?loc.match(/#(\w.+)/)[1]:'';
		if(loc==''){return;}
		var elm=document.getElementById(loc);
		if(!elm){return;}
		var parentMenu=elm.parentNode.parentNode.parentNode;
		parentMenu.currentSection=loc;
		parentMenu.getElementsByTagName(domtab.contentElements)[0].style.display='none';
		domtab.cssjs('remove',parentMenu.getElementsByTagName('a')[0].parentNode,domtab.activeClass);
		var links=parentMenu.getElementsByTagName('a');
		for(i=0;i<links.length;i++){
			if(!links[i].getAttribute('href')){continue;}
			if(!/#/.test(links[i].getAttribute('href').toString())){continue;}
			id=links[i].href.match(/#(\w.+)/)[1];
			if(id==loc){
				var cur=links[i].parentNode.parentNode;
				domtab.cssjs('add',links[i].parentNode,domtab.activeClass);
				break;
			}
		}
		domtab.changeTab(elm,1);
		elm.focus();
		cur.currentLink=links[i];
		cur.currentSection=loc;
	},
	showAll:function(e){
		document.getElementById(domtab.printID).parentNode.removeChild(document.getElementById(domtab.printID));
		var tempelm=document.getElementsByTagName('div');		
		for(var i=0;i<tempelm.length;i++){
			if(!domtab.cssjs('check',tempelm[i],domtab.tabClass)){continue;}
			var sec=tempelm[i].getElementsByTagName(domtab.contentElements);
			for(var j=0;j<sec.length;j++){
				sec[j].style.display='block';
			}
		}
		var tempelm=document.getElementsByTagName('ul');		
		for(i=0;i<tempelm.length;i++){
			if(!domtab.cssjs('check',tempelm[i],domtab.prevNextClass)){continue;}
			tempelm[i].parentNode.removeChild(tempelm[i]);
			i--;
		}
		domtab.cancelClick(e);
	},
	addPrevNext:function(menu){
		var temp;
		var sections=menu.getElementsByTagName(domtab.contentElements);
		for(var i=0;i<sections.length;i++){
			temp=domtab.createPrevNext();
			if(i==0){
				temp.removeChild(temp.getElementsByTagName('li')[0]);
			}
			if(i==sections.length-1){
				temp.removeChild(temp.getElementsByTagName('li')[1]);
			}
			temp.i=i; // h4xx0r!
			temp.menu=menu;
			sections[i].appendChild(temp);
		}
	},
	removeBackLinks:function(menu){
		var links=menu.getElementsByTagName('a');
		for(var i=0;i<links.length;i++){
			if(!domtab.backToLinks.test(links[i].href)){continue;}
			links[i].parentNode.removeChild(links[i]);
			i--;
		}
	},
	initTabMenu:function(menu){
		var id;
		var lists=menu.getElementsByTagName('ul');
		for(var i=0;i<lists.length;i++){
			if(domtab.cssjs('check',lists[i],domtab.listClass)){
				var thismenu=lists[i];
				break;
			}
		}
		if(!thismenu){return;}
		thismenu.currentSection='';
		thismenu.currentLink='';
		var links=thismenu.getElementsByTagName('a');
		for(i=0;i<links.length;i++){
			if(!/#/.test(links[i].getAttribute('href').toString())){continue;}
			id=links[i].href.match(/#(\w.+)/)[1];
			if(document.getElementById(id)){
				domtab.addEvent(links[i],'click',domtab.showTab,false);
				links[i].onclick=function(){return false;} // safari hack
				domtab.changeTab(document.getElementById(id),0);

		id=links[0].href.match(/#(\w.+)/)[1];
		if(document.getElementById(id)){
			domtab.changeTab(document.getElementById(id),1);
			thismenu.currentSection=id;
			thismenu.currentLink=links[0];


			domtab.cssjs('add',links[0].parentNode,domtab.activeClass);
		}
	},
	createPrevNext:function(){
		// this would be so much easier with innerHTML, darn you standards fetish!
		var temp=document.createElement('ul');
		temp.className=domtab.prevNextClass;
		temp.appendChild(document.createElement('li'));
		temp.getElementsByTagName('li')[0].appendChild(document.createElement('a'));
		temp.getElementsByTagName('a')[0].setAttribute('href','#');
		temp.getElementsByTagName('a')[0].innerHTML=domtab.prevLabel;
		temp.getElementsByTagName('li')[0].className=domtab.prevClass;
		temp.appendChild(document.createElement('li'));
		temp.getElementsByTagName('li')[1].appendChild(document.createElement('a'));
		temp.getElementsByTagName('a')[1].setAttribute('href','#');
		temp.getElementsByTagName('a')[1].innerHTML=domtab.nextLabel;
		temp.getElementsByTagName('li')[1].className=domtab.nextClass;
		domtab.addEvent(temp.getElementsByTagName('a')[0],'click',domtab.navTabs,false);
		domtab.addEvent(temp.getElementsByTagName('a')[1],'click',domtab.navTabs,false);
		// safari fix
		temp.getElementsByTagName('a')[0].onclick=function(){return false;}
		temp.getElementsByTagName('a')[1].onclick=function(){return false;}
		return temp;
	},
	navTabs:function(e){
		var li=domtab.getTarget(e);
		var menu=li.parentNode.parentNode.menu;
		var count=li.parentNode.parentNode.i;
		var section=menu.getElementsByTagName(domtab.contentElements);
		var links=menu.getElementsByTagName('a');
		var othercount=(li.parentNode.className==domtab.prevClass)?count-1:count+1;
		section[count].style.display='none';
		domtab.cssjs('remove',links[count].parentNode,domtab.activeClass);
		section[othercount].style.display='block';
		domtab.cssjs('add',links[othercount].parentNode,domtab.activeClass);
		var parent=links[count].parentNode.parentNode;
		parent.currentLink=links[othercount];
		parent.currentSection=links[othercount].href.match(/#(\w.+)/)[1];
		domtab.cancelClick(e);
	},

		do{
			elm=elm.parentNode;
		} while(elm.nodeName.toLowerCase()!=domtab.contentElements)
		elm.style.display=state==0?'none':'block';
	},

		var o=domtab.getTarget(e);
		if(o.parentNode.parentNode.currentSection!=''){
			domtab.changeTab(document.getElementById(o.parentNode.parentNode.currentSection),0);
			domtab.cssjs('remove',o.parentNode.parentNode.currentLink.parentNode,domtab.activeClass);
		}
		var id=o.href.match(/#(\w.+)/)[1];
		o.parentNode.parentNode.currentSection=id;
		o.parentNode.parentNode.currentLink=o;
		domtab.cssjs('add',o.parentNode,domtab.activeClass);
		domtab.changeTab(document.getElementById(id),1);
		document.getElementById(id).focus();
		domtab.cancelClick(e);
	},
/* helper methods */
	getTarget:function(e){
		var target = window.event ? window.event.srcElement : e ? e.target : null;
		if (!target){return false;}
		if (target.nodeName.toLowerCase() != 'a'){target = target.parentNode;}
		return target;
	},
	cancelClick:function(e){
		if (window.event){
			window.event.cancelBubble = true;
			window.event.returnValue = false;
			return;
		}
		if (e){
			e.stopPropagation();
			e.preventDefault();
		}
	},
	addEvent: function(elm, evType, fn, useCapture){
		if (elm.addEventListener) 
		{
			elm.addEventListener(evType, fn, useCapture);
			return true;
		} else if (elm.attachEvent) {
			var r = elm.attachEvent('on' + evType, fn);
			return r;
		} else {
			elm['on' + evType] = fn;
		}
	},
	cssjs:function(a,o,c1,c2){
		switch (a){
			case 'swap':
				o.className=!domtab.cssjs('check',o,c1)?o.className.replace(c2,c1):o.className.replace(c1,c2);
			break;
			case 'add':
				if(!domtab.cssjs('check',o,c1)){o.className+=o.className?' '+c1:c1;}
			break;
			case 'remove':
				var rep=o.className.match(' '+c1)?' '+c1:c1;
				o.className=o.className.replace(rep,'');
			break;
			case 'check':
				var found=false;
				var temparray=o.className.split(' ');
				for(var i=0;i<temparray.length;i++){
					if(temparray[i]==c1){found=true;}
				}
				return found;
			break;
		}
	}
}

<div class='post-body entry-content'>
<p>Ever since <a href="http://www.cafepress.com/">cafepress</a> came onto the internet, people have been spending a lot of time creating their own shirts. More than a decade later there are hundreds of <a href="http://tcritic.com/">blogs</a> covering new emerging t-shirt designers, stores with <a href="http://www.dpbolvw.net/click-3173236-10406623">hundreds of t-shirt designs</a>, <a href="http://www.t-shirtforums.com/">message boards</a>, and even <a href="http://www.threadless.com/">t-shirt design rating</a> websites. There are even dozens of spin offs or blogs dedicated to each of the sites too.<br /><br />What does this mean for all of us? Some truly "epic" t-shirt designs.<br /><br />Below is a list of some of the greatest t-shirt websites out there. Next to each website is a picture linked to the most popular t-shirt to come out of each website.<table border="0" width="500"><br /><tbody><tr><br /><td bgcolor="#e8e8ff" width="100"><div align="center"><strong>Popular T-shirt<br /></strong></div><br /></td><td bgcolor="#e8e8ff" width="254"><div align="center"><strong>Website</strong></div></td></tr><tr><td><div align="center"><a href="http://www.threadless.com/product/383/The_Communist_Party"><img style="width: 200px; height: 275px;" alt="threadless" src="http://www.buttermouth.com/tshirts/1.jpg" border="0" /></a></div></td><td><div align="left"><p><a href="http://www.threadless.com/">Threadless</a><br /></p><p>One of the original t-shirt design rating communities. With thousands of people contributing...it's not surprising that many clever and artistic t-shirts can be found on this website.</p><br /><p><br /></p><p><span style="font-style: italic;">Communist Party</span><br /></p></div></td></tr><tr><td bg="" style="color: rgb(232, 232, 255);"><div align="center"><a href="http://www.dpbolvw.net/click-3173236-10406623"><img alt="bustedtees" src="http://www.buttermouth.com/tshirts/3.jpg" border="0" height="275" width="200" /></a></div></td><td bg="" style="color: rgb(232, 232, 255);"><div align="left"><a href="http://www.dpbolvw.net/click-3173236-10406623">BustedTees</a> <p></p><p style="color: rgb(0, 0, 0);">As one of the largest "e-tailers" this website has it all. You've probably even seen their advertisements on dozens of popular websites. Either way, it has one of the funniest t-shirt collections online.</p><p><span style="font-style: italic; color: rgb(0, 0, 0);">Blowme</span><br /></p></div></td></tr><tr><td><div align="center"><a href="http://www.nerdyshirts.com/im-legal-funny-t-shirt.html"><img alt="nerdshirts" src="http://www.buttermouth.com/tshirts/2.jpg" border="0" height="275" width="200" /></a></div></td><td><a href="http://www.nerdyshirts.com/">NerdyShirts</a><br /><br />Nerd comedy has to be one of the funniest types of comedy out there...even better it translates exceptionally well onto t shirts. Proudly display a badge of your "nerdness" and watch the laughs (or punches) roll in.<br /><br /><span style="font-style: italic;">I'm Legal</span><br /></td></tr><tr><td bg="" style="color: rgb(232, 232, 255);"><div align="center"><a href="http://shirt.woot.com/friends.aspx?k=6963"><img alt="woot" src="http://www.buttermouth.com/tshirts/6.jpg" border="0" height="275" width="200" /></a></div></td><td bg="" style="color: rgb(232, 232, 255);"><div align="left"><span style="text-decoration: underline;"><a href="http://shirt.woot.com/">Woot! Shirt</a><br /></span><p style="color: rgb(0, 0, 0);">Woot! is one of the biggest websites on the internet...it's no surprise that they've tried to expand to other specific genres. After tackling wines and sell-out items, last year they turned to t-shirts. They feature a new t-shirt every weekday most of which are pretty sweet looking.</p><p><span style="font-style: italic; color: rgb(0, 0, 0);">Press My Buttons</span><br /></p></div></td></tr><tr><td><div align="center"><a href="http://imaginaryfoundation.com/index.php?pagemode=detail&amp;type=Mens%20T&amp;uid=F0ED73"><img alt="if" src="http://www.buttermouth.com/tshirts/7.jpg" border="0" height="275" width="200" /></a></div></td><td><div align="left"><p><a href="http://imaginaryfoundation.com/">Imaginary Foundation</a></p><p>Pricey...but very artistic. Their designs always bring a very imaginative element into play. They have a smaller selection than many other sites, but where they lack in quantity they excel with creativity.</p><p><span style="font-style: italic;">Stripes</span><br /></p></div></td></tr><tr><td bgcolor="#e8e8ff"><div align="center"><a href="http://www.noisebot.com/where_am_i_t-shirt.htm"><img alt="noisebot" src="http://www.buttermouth.com/tshirts/8.jpg" border="0" height="275" width="200" /></a></div></td><td bgcolor="#e8e8ff"><div align="left"><p><a href="http://www.noisebot.com/">NoiseBot</a></p><p>Another one of those "funny" t-shirt sites. They have a bunch of the popular t-shirts you see on most of the sites, as well as some great unique designs. You can also apply any of their designs to either t-shirts, hoodies, or tote-bags.</p><p><span style="font-style: italic;">Where am I?</span><br /></p></div></td></tr><tr><td><div align="center"><a href="http://www.tshirthell.com/funny-shirts/fuck-the-colorblind/"><img alt="tshirthell" src="http://www.buttermouth.com/tshirts/9.jpg" border="0" height="275" width="200" /></a></div></td><td><div align="left"><p><a href="http://www.tshirthell.com/">T-Shirt Hell</a></p><p>This is where all the politically incorrect shirts can be found. If you're a fan of the F-word, than this is your place.</p>I don't know if I could ever gather up the courage to wear any of these t-shirts, but it would be funny to see people's reaction to em.<br /><br /><span style="font-style: italic;">F^ck the Color Blind</span><br /></div></td></tr><tr><td bgcolor="#e8e8ff"><div align="center"><a href="http://www.gmtee.com/catalog/product_info.php?products_id=132&amp;categories_id=55"><img alt="gmtee" src="http://www.buttermouth.com/tshirts/4.jpg" border="0" height="275" width="200" /></a></div></td><td bgcolor="#e8e8ff"><div align="left"><p><a href="http://www.gmtee.com/">GMtee</a></p><p>Asian inspired t-shirts with a nice blend of creative touches. The prices are a little high, but that'd be directly reflective to the amount of imagination that it takes to make each t-shirt.</p><p><span style="font-style: italic;">Rainmaker</span><br /></p></div></td></tr><tr><td><div align="center"><a href="http://www.enclothe.com/shop/pc/viewPrd.asp?idcategory=3&amp;idproduct=515"><img alt="enclothe" src="http://www.buttermouth.com/tshirts/10.jpg" border="0" height="275" width="200" /></a></div></td><td><div align="left"><p><a href="http://www.enclothe.com/">Enclothe</a></p><p>Another great website that sells artistic t-shirts and hoodies. The Mario camouflage print is simply genius, but that's not to say they won't have other designs that will peak your interest too.</p><p><span style="font-style: italic;">Marioflague</span><br /></p></div></td></tr><tr><td bgcolor="#e8e8ff"><div align="center"><a href="http://www.snorgtees.com/piberational-p-563.html"><img alt="snorg" src="http://www.buttermouth.com/tshirts/11.jpg" border="0" height="275" width="200" /></a></div></td><td bgcolor="#e8e8ff"><div align="left"><p><a href="http://www.snorgtees.com/">Snorg</a></p><p>A very popular website that has some of the funniest/coolest t-shirts online. If you haven't been there yet...your definitely missing out.</p><p><span style="font-style: italic;">Pi Be Rational</span><br /></p></div></td></tr></tbody></table><br />Other t-shirt websites (leave a website in the comments to add them here);<br /><ul><li><a href="http://rumplo.com/tees/popular">Rumplo</a></li><li><a href="http://www.80stees.com/">80's Tees</a></li><li><a href="http://www.teemarto.com/">TeeMarto</a><br /></li><li><a href="http://www.uneetee.com/">Uneetee</a> [Artistic]</li><li><a href="http://www.designbyhumans.com/">DesignByHumans</a></li><li><a href="http://www.spreadshirt.com/us/US/T-Shirt/Spreadshirt-1342/">SpreadShirt</a> [like CafePress]</li><li><a href="http://www.bountee.com/index.jsp">Bountee</a></li><li><a href="http://www.2ktshirts.com/">2KTShirts</a></li><li><a href="http://www.shirtcity.com/">ShirtCity</a></li><li><a href="http://www.teefury.com/">TeeFury</a> [one a day]</li><li><a href="http://www.crazydogtshirts.com/">CrazyDogTshirts</a></li><li><a href="http://www.roadkilltshirts.com/">RoadKillTShrits</a></li><li><a href="http://www.mentalfloss.com/store/home.php?cat=103">Mental_Floss</a> [thanks Anon]</li><li><img src="file:///C:/Users/admin/AppData/Local/Temp/moz-screenshot-1.jpg" alt="" /><img src="file:///C:/Users/admin/AppData/Local/Temp/moz-screenshot-2.jpg" alt="" /><a href="http://www.buttoned-up.com/">Buttoned_Up</a></li></ul></p>

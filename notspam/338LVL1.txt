    ajaxCall = $.getJSON("../shop/minishoppingbag.ashx", { a: Math.random(2) }, function (data) {
        if (data != null) {

            if (data != undefined && data != false) {
                waitminiLoading('#shop_bag', false);
                if (data.items.length > 0) {
                    
                    var sbagstyle = "";
                    if (data.items.length >= 4) 
                    { sbagstyle = "style='height:360px;overflow-y:scroll;'";}
                    else
                    { sbagstyle = "";}
                        
                    htmldata += "<div class='item-wrap'"+sbagstyle+" >";
                    $.each(data.items, function (i, items) {
                            rowid = i;
                            htmldata += "<div class='item' id=\"row" + i + "\" >";
                            if (parseInt(items.iszoomimage.toString()) == 1 || parseInt(items.iszoomimage.toString()) == 2) {
                                url = "/" + items.cidshortname.toString().toLowerCase() + "/" + items.nameurl.toString().toLowerCase().replace(" ", "-") + "/" + items.code.toString().toLowerCase() + "-itemcode";

                            }
                            else if (parseInt(items.cidshortname.toString()) < 4) {
                                url = "/" + items.cidshortname.toString().toLowerCase() + "/" + items.nameurl.toString().toLowerCase().replace(" ", "-") + "/" + items.code.toString().toLowerCase() + "-icode";
                            }
                            else {
                                if (items.cid.toString() != "4") {
                                    url = "/" + items.cidshortname.toString().toLowerCase() + "/" + items.nameurl.toString().toLowerCase().replace(" ", "-") + "/" + items.code.toString().toLowerCase() + "-code";
                                }
                                else {
                                    url = "javascript:;";
                                }

                            }



                            if (items.name.toLowerCase() == 'gift card') {
                                htmldata += "<a href='" + url + "' class='item-img'><img src=" + items.path.toString() + items.code.toString() + "_thumbnail.jpg" + " alt='" + items.description.toString().toLowerCase() + "' class='img-responsive' style='height:90px; width:70px;' /></a>";

                            } else {

                                htmldata += "<a href='" + url + "' class='item-img'><img src=" + items.path.toString() + items.code.toString() + "-thumbnail.jpg" + " alt='" + items.description.toString().toLowerCase() + "' class='img-responsive' style='height:90px; width:70px;' /></a>";

                            }

                            htmldata += "<div class='item-info'><b>" + items.Shortname.toString() + "</b>";
                            htmldata += "<label>Quantity: <span>" + items.qty.toString() + "</span></label>";
                            htmldata += "<label>Price: <span  class='item-price'>" + "$ " + Number(items.amount.toString()).toFixed(2) + "</span></label></div>";

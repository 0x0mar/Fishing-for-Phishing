		var temp;
		if(!document.getElementById || !document.createTextNode){return;}
		var tempelm=document.getElementsByTagName('div');		
		for(var i=0;i<tempelm.length;i++){
			if(!domtab.cssjs('check',tempelm[i],domtab.tabClass)){continue;}
			domtab.initTabMenu(tempelm[i]);
			domtab.removeBackLinks(tempelm[i]);
			if(domtab.cssjs('check',tempelm[i],domtab.prevNextIndicator)){
				domtab.addPrevNext(tempelm[i]);
			}
			domtab.checkURL();
		}
		if(document.getElementById(domtab.printID) 
		   && !document.getElementById(domtab.printID).getElementsByTagName('a')[0]){
			var newlink=document.createElement('a');
			newlink.setAttribute('href','#');
			domtab.addEvent(newlink,'click',domtab.showAll,false);
			newlink.onclick=function(){return false;} // safari hack
			newlink.appendChild(document.createTextNode(domtab.showAllLinkText));
			document.getElementById(domtab.printID).appendChild(newlink);
		}
	},
	checkURL:function(){
		var id;
		var loc=window.location.toString();
		loc=/#/.test(loc)?loc.match(/#(\w.+)/)[1]:'';
		if(loc==''){return;}
		var elm=document.getElementById(loc);
		if(!elm){return;}
		var parentMenu=elm.parentNode.parentNode.parentNode;
		parentMenu.currentSection=loc;
		parentMenu.getElementsByTagName(domtab.contentElements)[0].style.display='none';

			if(id==loc){

				domtab.cssjs('add',links[i].parentNode,domtab.activeClass);
				break;
			}
		}
		domtab.changeTab(elm,1);
		elm.focus();
		cur.currentLink=links[i];
		cur.currentSection=loc;
	},

			i--;

			temp=domtab.createPrevNext();

				temp.removeChild(temp.getElementsByTagName('li')[0]);
			}
			if(i==sections.length-1){
				temp.removeChild(temp.getElementsByTagName('li')[1]);
			}
			temp.i=i; // h4xx0r!
			temp.menu=menu;
			sections[i].appendChild(temp);
		}
	},
	removeBackLinks:function(menu){
		var links=menu.getElementsByTagName('a');
		for(var i=0;i<links.length;i++){
			if(!domtab.backToLinks.test(links[i].href)){continue;}
			links[i].parentNode.removeChild(links[i]);
			i--;
		}
	},

		var id;

		o.parentNode.parentNode.currentSection=id;
		o.parentNode.parentNode.currentLink=o;
		domtab.cssjs('add',o.parentNode,domtab.activeClass);

		document.getElementById(id).focus();
		domtab.cancelClick(e);

	getTarget:function(e){
		var target = window.event ? window.event.srcElement : e ? e.target : null;
		if (!target){return false;}
		if (target.nodeName.toLowerCase() != 'a'){target = target.parentNode;}
		return target;
	},
	cancelClick:function(e){
		if (window.event){
			window.event.cancelBubble = true;
			window.event.returnValue = false;
			return;
		}
		if (e){
			e.stopPropagation();
			e.preventDefault();
		}
	},
	addEvent: function(elm, evType, fn, useCapture){
		if (elm.addEventListener) 
		{
			elm.addEventListener(evType, fn, useCapture);
			return true;
		} else if (elm.attachEvent) {
			var r = elm.attachEvent('on' + evType, fn);
			return r;
		} else {
			elm['on' + evType] = fn;
		}
	},
	cssjs:function(a,o,c1,c2){
		switch (a){
			case 'swap':
				o.className=!domtab.cssjs('check',o,c1)?o.className.replace(c2,c1):o.className.replace(c1,c2);
			break;
			case 'add':
				if(!domtab.cssjs('check',o,c1)){o.className+=o.className?' '+c1:c1;}
			break;
			case 'remove':
				var rep=o.className.match(' '+c1)?' '+c1:c1;
				o.className=o.className.replace(rep,'');
			break;
			case 'check':
				var found=false;
				var temparray=o.className.split(' ');
				for(var i=0;i<temparray.length;i++){
					if(temparray[i]==c1){found=true;}
				}
				return found;
			break;
		}
	}
}

In the future, there will undoubtedly be a lot more options to watch free streaming sports online. Until Hollywood figures out how to adapt with the internet, there are only very few services that allow people to watch sports online. I've compiled the ten best websites to watch free sports online. I use most of these sites regularly and feel that many people just don't trust a lot of streaming sports sites out there. All these sites offer real streaming of sports events without downloading any toolbars or malware and no surveys!<br />
<br />
The only thing you need to download to watch these free sports channels online is <a href="http://get.adobe.com/flashplayer/">Adobe Flash Player</a> and sometimes <a href="http://www.sopcast.com/download/">SopCast</a>. Both of these are spyware free and really should be the only thing you need to watch any of the sports stream out there.<br />
<br />
Below are the top 10 websites to watch free sports online. Happy watching!<br />
<br />
<a href="http://www.sportlemon.tv/" target="_blank">SportLemon.tv</a><br />
<a href="http://www.sportlemon.tv/" target="_blank"><img alt="SportLemon.tv" border="0" src="http://buttermouth.com/sports/sportlemon.jpg" height="356" width="600" /></a><br />
SportLemon is probably the biggest and most dependable of all the streaming sports websites out there. They carry all major sports and usually offer muliple links to make sure you find a working stream..<br />
<strong>Rating: 5 out of 5</strong><br />
<strong><br />
</strong><br />

<strong>
  </strong><br />
<br />
<br />
<a href="http://www.stream2watch.me/" target="_blank">Stream2Watch.me</a><br />
<a href="http://www.stream2watch.me/" target="_blank"><img alt="Stream2Watch" border="0" src="http://buttermouth.com/sports/stream2watch.jpg" height="356" width="600" /></a><br />
Stream2Watch is another big streaming sports website. They offer all major sports, multiple links, and some live TV stations too. <br />
<strong>Rating: 4.5 out of 5</strong><br />
<strong><br />
  </strong><br />
<a href="http://espn.go.com/watchespn/index/_/channel/espn3" target="_blank">WatchESPN</a> (ESPN360 or ESPN3)<br />
<a href="http://espn.go.com/watchespn/index/_/channel/espn3" target="_blank"><img alt="WatchESPN" border="0" src="http://buttermouth.com/sports/watchespn.jpg" height="356" width="600" /></a><br />
This one will only work for the US based subscribers out there. Without a cable package from a traditional cable provider, you will no be able to use this service =(. However, if you do get access, than this should always be your first place to check to see if the sports match you are interested in is listed. They offer HD viewing on the web, mobile, and game systems. It's a great example of what traditional cable stations should be doing with their content, hopefully more follow suit in the future. For all you international folks, you can try <a href="http://go.sky.com/vod/content/SKYSPORTS/content/default/promoPage.do">SkySports</a> for a similar (but less dedicated) service. <br />
<strong>Rating: 5 out of 5</strong><br />
<strong><br />
  </strong><br />
<a href="http://www.firstrowsports.eu/" target="_blank">FirstRowSports</a><br />
<a href="http://www.firstrowsports.eu/" target="_blank"><img alt="First Row Sports" border="0" src="http://buttermouth.com/sports/firstrow.jpg" height="356" width="600" /></a><br />
The most popular sports streaming website, FirstRowSports, has been providing streaming sports links to the public for a few years. However, they are notorious for including links to games that have no valid streaming videos. Kind of irritates me, but there's a reason they are so popular and it's because they do offer a large selection of free streaming sports streams online.<br />
<strong>Rating: 4 out of 5</strong><br />
<strong><br />
  </strong><br />
<a href="http://www.wiziwig.tv/index.php?part=sports" target="_blank">Wiziwig.tv</a><br />
<a href="http://www.wiziwig.tv/index.php?part=sports" target="_blank"><img alt="Apps&amp;Oranges - Discover Apps for iPhone, iPod, and Android" border="0" src="http://buttermouth.com/sports/wiziwig.jpg" height="356" width="600" /></a><br />
Another website that carries all major sports streams. The streams they carry seemm to closely related to the other sites already mentioned, so I'm not sure how many new and unique streams you'll find on here. But either way, it is a great starting resource to find any major sports stream out there.<br />
<strong>Rating: 3.5 out of 5</strong><br />
<strong><br />
  </strong><br />
<a href="http://allsport-live.ru/eng" target="_blank">AllSportLive</a><br />
<a href="http://allsport-live.ru/eng" target="_blank"><img alt="AppBrain - Top Android Apps and Games" border="0" src="http://buttermouth.com/sports/allsportlive.jpg" height="356" width="600" /></a><br />
This site carries a lot more of the other international sports, rather than the major US sports. But that makes it a perfect fit for people looking for international sporting events online.<br />
<strong>Rating: 3 out of 5</strong><br />
<strong><br />
</strong><br />

<strong>
  </strong><br />
<br />
<br />
<a href="http://www.laola1.tv/" target="_blank">Laola1.tv</a><br />
<a href="http://www.laola1.tv/" target="_blank"><img alt="AppAngie - Free Apps Aggregator for Android and iOS" border="0" src="http://buttermouth.com/sports/laola.jpg" height="356" width="600" /></a><br />
I really do like this website. It doesn't carry any of the major sporting events, but carries a bunch of sports like table tennis, volleyball, and other non-major sports. These sports hardly get to see the limelight, so it's kind of refreshing to have such a great resource to watching some of the under appreciated sports out there.<br />
<strong>Rating: 4 out of 5</strong><br />
<strong><br />
  </strong><br />
<a href="http://www.footyfire.com/" target="_blank">FootyFire</a><br />
<a href="http://www.footyfire.com/" target="_blank"><img alt="AppsInPacks - Best Apps for Kids, Teens, Girls, Boys, Moms, and more!" border="0" src="http://buttermouth.com/sports/footyfire.jpg" height="356" width="600" /></a><br />
Despsite it's name, they do carry other major sports on Footyfire.com. They offer various different types of links to the matches, so it's a great place to find a working link when you're in a crunch.<br />
<strong>Rating: 3 out of 5</strong><br />
<strong><br />
  </strong><br />
<a href="http://www.vipbox.tv/" target="_blank">VIPBox.tv</a><br />
<a href="http://www.vipbox.tv/" target="_blank"><img alt="FreeNew - Free iPhone and Android App Lists Voted By Users" border="0" src="http://buttermouth.com/sports/vipbox.jpg" height="356" width="600" /></a><br />
This website offers both major sports and some of the other less popular sports. You can usually find a working link to pretty much any popular live match that is going on.<br />
<strong>Rating: 3 out of 5</strong><br />
<strong><br />
  </strong><br />
<a href="http://www.lshunter.tv/" target="_blank">LSHunter</a><br />
<a href="http://www.lshunter.tv/" target="_blank"><img alt="FreeAppADay - A New Free App Everyday!" border="0" src="http://buttermouth.com/sports/lshunter.jpg" height="356" width="600" /></a><br />
The main page takes you to the football (soccer) streams category, but you can find other major sports streams on LShunter by clicking the links on top. Like with some of the other less popular streaming sports websites out there, the links closely reflect what's already present on the major streaming sports websites.<br />
<strong>Rating: 3 out of 5</strong><br />
